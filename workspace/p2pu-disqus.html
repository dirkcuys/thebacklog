# Convert all the XML posts to python dictionaries
json_data = []
for post in posts:
    post_data = {}
    post_data['date'] = post.find('{http://disqus.com}createdAt').text
    author = post.find('{http://disqus.com}author')
    post_data['author_email'] = author.find('{http://disqus.com}email').text
    post_data['author_name'] = author.find('{http://disqus.com}name').text
    post_data['message'] = post.find('{http://disqus.com}message').text
    post_data['thread_id'] = post.find('{http://disqus.com}thread').attrib['{http://disqus.com/disqus-internals}id']
    #thread = root.find('{http://disqus.com}thread[@{http://disqus.com}id="' + post_data['thread_id'] + '"]')
    post_data['link'] = root.xpath('//d:thread[@dsq:id="' + post_data['thread_id'] + '"]/d:link', namespaces={'d': 'http://disqus.com', 'dsq': 'http://disqus.com/disqus-internals'})[0].text
    post_data['post_id'] = post.attrib['{http://disqus.com/disqus-internals}id']
    if post.find('{http://disqus.com}parent') is not None:
        post_data['parent_id'] = post.find('{http://disqus.com}parent').attrib['{http://disqus.com/disqus-internals}id']
    else:
        post_data['parent_id'] = ''
    json_data += [post_data]

ff = lambda x: x['link'].startswith('https://p2pu.org/')
json_data = filter(ff, json_data)

# function to dump data to a file
import unicodecsv
def write_to_csv(data, filename):
    """ data should be a list of dicts """
    if len(data) == 0:
        return
    with open(filename, 'w') as f:
        writer = unicodecsv.writer(f)
        writer.writerow(data[0].keys())
        for elem in data:
            writer.writerow(elem.values())

# Lets get comment by course
regex = r'^https://p2pu.org/(?P<lang>.*)/courses/(?P<course_id>\d+)/.*'
import re

def f(link):
    match = re.match(regex, link)
    if match:
        return 'https://p2pu.org/' + match.group('lang') + '/courses/' + match.group('course_id') + '/'
    raise Exception('Invalid course URL: ' + link)

data['course_id'] = data['link'].apply(f)

gb = data.groupby(['course_id']).count()
